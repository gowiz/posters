  }

    \LARGE
    \begin{align*}

      -(\pm a \pm b) &= \mp a \mp b
    \end{align*}
  }

  \block{Breuken}{
    \LARGE
    \begin{align*}
      \dfrac{a+b}{c} &= \dfrac{a}{c} + \dfrac{b}{c}\\
      \dfrac{a}{b} + \dfrac{c}{d} &= \dfrac{ad + bc}{bd}\\
      a\cdot \dfrac{1}{b} &= \dfrac{a}{b}\\
      a\cdot \dfrac{b}{c} &= \dfrac{ab}{c}\\
      \dfrac{a}{b} \cdot \dfrac{c}{d} &= \dfrac{ac}{bd}\\
      \dfrac{ac}{bc} &= \dfrac{a}{b}\\
      -\dfrac{a}{b} &= \dfrac{-a}{b} = \dfrac{a}{-b}\\
      \left. \dfrac{a}{b} \middle/ \dfrac{c}{d} \right. &= \dfrac{a}{b}\cdot \dfrac{d}{c} = \dfrac{ad}{bc}
    \end{align*}
  }
  \block{Niet wat je denkt}{
    \LARGE
    \begin{align*}
      \dfrac{a}{b+c} &\neq \cdots
    \end{align*}
  }

  \block{Machten}{
    \LARGE
    \begin{align*}
      a^m\cdot a^n &= a^{m+n}\\
      \dfrac{a^m}{a^n} &= a^{m-n}\\
      \left(a^m\right)^{^n} &= a^{m\cdot n}\\
      \left(ab\right)^m &= a^mb^m\\
      \left(\dfrac{a}{b}\right)^m &= \dfrac{a^m}{b^m}\\
      a^{-n} &= \dfrac{1}{a^n}\\
      a^{n/m} &= \sqrt[m]{a^n}
    \end{align*}
  }
  \block{Teken}{

    \end{align*}
  }
